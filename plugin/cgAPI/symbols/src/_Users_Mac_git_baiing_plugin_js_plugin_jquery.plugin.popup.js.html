692  * @author 徐晨 
693  * @name jQuery.cgResultTips
694  * @class 提示框 
695  * @constructor
696  * @extends jQuery
697  * @since version 1.0
698  * @param {String} options.title 弹出框的标题
699  * @param {String} options.type 弹窗模式 [message | error]
700  * @param {String|Array} options.message 显示的消息提示文本 type为message时为字符串类型，type为error时为字符串数组类型
701  * @param {Number} options.time type为message时自动关闭窗口时间
702  * @example $.cgResultTips({
703   [可选]title: "提示",
704         type : "message|error",
705         message : [message : "string" | error : ["array"]],
706   [可选]time : 2000

709 $.cgResultTips = function(options){
710     var option = $.extend({
711         title : "",
712         type : "",
713         message : [],
714         time : 3000
715     }, options);
716     var str = "";
717     switch(option.type){
718         case "message" :
719             if ( typeof(option.message) == "string" ) {
720                 str = '<p class="confirm-msg Js-popup-msg">' + option.message + '</p>';
721             }
722             option.title = option.title ? option.title : "操作成功";
723             option.con = function(opt){
724                 setTimeout(function(){
725                     opt.oPopup.fadeOut("normal", function(){
726                         opt.close();
727                     });
728                 }, option.time);
729             };
730             break;
731         case "error" :
732             if ( option.message.constructor === Array && option.message.length > 0 ) {
733                 for( var i = 0; i < option.message.length; i++ ){
734                     str += '<p class="confirm-error-msg Js-popup-msg">' + option.message[i] + '</p>';
735                 }
736             }
737             option.title = option.title ? option.title : "操作失败";
738             option.con = function(opt){
739 
740             };
741             break;
742         default :
743             break;
744     };
745     $.cgPopup({
746         title : option.title,
747         template : str,
748         isLayer : true,
749         hasBtn : true,
750         hasCancel : false,
751         content : function(opt){
752             option.con(opt);
753         },
754         done : function(opt){
755             opt.close();
756         }
757     });
758 };

816 $.cgPopup = function(options){
817     var option = $.extend({
818         title: "提示",
819         template : "",
820         addClass : "",
821         isLayer : true,
822         isCenter : true,
823         isOnly : false,
824         append : {
825             isAppend : false,
826             dom : $(".dom")
827         },
828         // hasError : false,
829         hasBtn : true,
830         hasCancel : true,
831         content : function(option){},
832         done : function(option){},
833         cancel : function(option){}
834     }, options);
835     if ( option.template == "" ) {
836         return false;
837     }
838     var $popup = $('<div class="module-popup fn-clear Js-popup-wrap">'+
839                   // '<div class="popup-wrap fn-clear">'+
840                     '<a href="javascript:void(0)" class="popup-close Js-popup-close"></a> '+
841                     '<h6 class="fn-clear popup-tit Js-popup-title">' + option.title + '</h6>'+
842                     '<div class="popup-con Js-popup-con"></div>'+
843                     '<div class="btn-wrap Js-popup-btn-wrap">'+
844                         // '<input type="button" class="popup-btn input-btn Js-popup-done" value="确定">'+
845                         // '<input type="button" class="popup-btn input-btn Js-popup-cancel" value="取消">'+
846                         '<a class="popup-btn Js-popup-done" href="javascript:;;"><span class="popup-btn-text">确认</span></a>'+
847                         '<a class="popup-btn Js-popup-cancel" href="javascript:;;"><span class="popup-btn-text">取消</span></a>'+
848                     '</div>'+
849                   // '</div>'+
850                 '</div>');
851     var $error = $('<ul class="result-tips-error result-tips"></ul>');
852     if ( option.isLayer ) {
853         var $layer = $("<div class='module-popup-layer'></div>");
854         $("body").append($layer);
855     }
856     if ( option.isOnly ) {
857         $(".module-popup-layer").remove();
858         $(".Js-popup-wrap").remove();
859     }
860     if ( option.append.isAppend ) {
861         option.append.dom.append( $popup );
862     } else{
863         $("body").append( $popup );
864     }
865     // if ( option.hasError ) {
866     //     $popup.find(".Js-popup-btn-wrap").after($error);
867     // }
868     // add new class
869     if ( option.addClass ) {
870         $popup.addClass(option.addClass);
871     }
872     var $mainWrap = $popup.find(".Js-popup-main-wrap"),
873         $btnWrap = $popup.find(".Js-popup-btn-wrap"),
874         $btnDone = $btnWrap.find(".Js-popup-done"),
875         $btnCancel = $btnWrap.find(".Js-popup-cancel"),
876         $btnClose = $popup.find(".Js-popup-close"),
877         $con = $popup.find(".Js-popup-con"),
878         positionCenter = function(){
879             var objWidht = $popup.width(),
880                 objHeight = $popup.height();
881             $popup.css( { "margin-left" : "-" + objWidht / 2 + "px" } );
882         };
883     // out param
884     $.extend(option, {
885         oPopup : $popup,
886         oBtnWrap : $btnWrap,
887         oBtnDone : $btnDone,
888         oBtnCancel : $btnCancel,
889         oBtnClose : $btnClose,
890         oCon : $con,
891         oTip : $error,
892         /**
893          * @name jQuery.cgPopup#close
894          * @desc  关闭弹出框
895          * @event
896          * @param {Function} callback 回调函数
897          */
898         close : function(){
899             $popup.remove();
900             if ( option.isLayer ) {
901                 $layer.remove();
902             }
903         },
904         /**
905          * @name jQuery.cgPopup#showTip
906          * @desc  显示错误信息
907          * @event
908          * @param {String} str 错误信息的字符串
909          */
910         showTip : function(str){
911             $mainWrap.append($error);
912             $error.html(str);
913         },
914         /**
915          * @name jQuery.cgPopup#removeTip
916          * @desc  隐藏错误信息
917          * @event
918          */
919         removeTip : function(str){
920             $error.remove();
921         },
922         /**
923          * @name jQuery.cgPopup#disableBtn
924          * @desc  改变确定按钮状态
925          * @event
926          * @param {Boolean} disable 为true时确定按钮不可用，反之可用
927          */
928         disableBtn : function(disable){
929             if (disable) {
930                 option.oBtnDone.prop("disabled", true);
931             }else{
932                 option.oBtnDone.prop("disabled", false);
933             }
934         }
935     });
936     // append content template
937     $con.append(option.template);
938 
939     $popup.show();
940     // bind popup init function
941     if ( typeof(option.content) ) {
942         option.content(option);
943     }
944 
945     if ( option.hasBtn ) {
946         $btnDone.on("click", function(){
947             if ( typeof(option.done) == "function" && !$(this).prop("disabled") ) {
948                 option.done(option);
949             }
950         });
951         if ( option.hasCancel ) {
952             $btnCancel.on("click", function(){
953                 if ( typeof(option.cancel) == "function" ) {
954                     option.cancel(option);
955                 }
956                 option.close();
957             });
958             $btnClose.on("click", function(){
959                 $btnCancel.click();
960             });
961         } else{
962             $btnCancel.remove();
963             $btnClose.on("click", function(){
964                 option.close();
965             });
966         }
967     }else{
968         $btnWrap.remove();
969         $btnClose.on("click", function(){
970             if ( typeof(option.done) == "function" ) {
971                 option.done(option);
972             }
973             option.close();
974         });
975     }
976 
977     // bind wrap position
978     if ( option.isCenter ) {
979         positionCenter();
980         $(window).resize(function(){
981             positionCenter();
982         });
983     }
984 };
985 
986 
987 }(jQuery));
