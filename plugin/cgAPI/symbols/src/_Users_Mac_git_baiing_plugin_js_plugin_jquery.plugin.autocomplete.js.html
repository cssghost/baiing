109 $.cgAutoComplete = function(options){
110     var option = {
111         frameBox : $(".Js-auto-frame"),
112         autoList : ".Js-auto-list",
113         autoItem : ".Js-auto-item",
114         autoText : ".Js-auto-text",
115         autoBtn : $(".Js-auto-btn"),
116         hoverClass : "selected",
117         titleClass : "",
118         isSearch : true,
119         init : function(){},
120         action : function(option, val){},
121         createSearchList : function(option, val){}
122     };
123     $.extend(option, options);
124     // 添加外部调用函数（受保护）
125     $.extend(option, {
126         _returnText : function(value){
127             option.text.val( value );
128         },
129         addItem : function($temp, data){
130             option.list.append( $.getFromTemplate( $temp, data ) );
131         },
132         showList : function(){
133             option.list.show();
134             $(document).one("click", function(){
135                 option.list.hide();
136             });
137         }
138     });
139     var keys = {
140         back: 8,
141         enter:  13,
142         escape: 27,
143         up:     38,
144         down:   40,
145         array:  [13, 27, 38, 40]
146     },keyDate = [], chooseData = [];
147 
148     var $list = option.frameBox.find(option.autoList),
149         $text = option.frameBox.find(option.autoText);
150     $.extend(option, { list : $list, text : $text });
151     $list.on("mouseenter", option.autoItem, function(){
152         $(this).addClass(option.hoverClass);
153         option._returnText( $(this).children().attr("title") );
154     }).on("mouseleave", option.autoItem,function(){
155             $(this).removeClass(option.hoverClass);
156         });
157 
158     $list.on("click", option.autoItem, function(){
159         option._returnText( $(this).children().attr("title") );
160         $list.hide();
161         if ( typeof( option.action ) == "function" ) {
162             option.action(option, $.trim( $text.val() ) );
163         }
164     });
165 
166     $text.off("keyup").on("keyup", function(event){
167         var keyCode = event.keyCode;
168         keyDate.push((new Date()).getTime());
169         var thisTime = keyDate[(keyDate.length - 1)];
170         var val = "";
171         // 需要检索列表
172         if (option.isSearch) {
173             if($.inArray(keyCode, keys.array) !=-1){
174                 if( $list.is(":visible") ){
175                     var $item = $list.children(option.autoItem),
176                         $cur = $item.filter("." + option.hoverClass);
177                     switch (keyCode){
178                         case keys.up:
179                             if ( $cur.length ){
180                                 $item.removeClass(option.hoverClass);
181                                 if ( $cur.index() == 1 ) {
182                                     $item.last().addClass(option.hoverClass);
183                                 } else{
184                                     if ( $cur.prev().hasClass(option.titleClass) ) {
185                                         $cur.prev().prev().addClass(option.hoverClass);
186                                     } else{
187                                         $cur.prev().addClass(option.hoverClass);
188                                     }
189                                 }
190                             }else{
191                                 $item.last().addClass(option.hoverClass);
192                             }
193                             option._returnText( $item.filter("." + option.hoverClass).children().attr("title") );
194                             break;
195                         case keys.down:
196                             if ( $cur.length ){
197                                 $item.removeClass(option.hoverClass);
198                                 if ( $cur.index() == $item.length ) {
199                                     $item.first().addClass(option.hoverClass);
200                                 } else{
201                                     if ( $cur.next().hasClass(option.titleClass) ) {
202                                         $cur.next().next().addClass(option.hoverClass);
203                                     } else{
204                                         $cur.next().addClass(option.hoverClass);
205                                     }
206                                 }
207                             }else{
208                                 $item.first().addClass(option.hoverClass);
209                             }
210                             option._returnText( $item.filter("." + option.hoverClass).children().attr("title") );
211                             break;
212                         case keys.enter:
213                             $list.hide();
214                             if ( typeof( option.action ) == "function" ) {
215                                 option.action(option, $.trim( $text.val() ) );
216                             }
217                             break;
218                         default :
219                             break;
220                     }
221                 }else{
222                     if( keyCode == keys.enter ){
223                         if ( typeof( option.action ) == "function" ) {
224                             option.action(option, $.trim( $text.val() ) );
225                         }
226                     }
227                 }
228             }else{
229                 // 延时显示searchlist 减少多余查询
230                 setTimeout(function(){
231                     val = $text.val();
232                     if( thisTime == keyDate[(keyDate.length - 1)] && val != "" ) {
233                         option.createSearchList(option, val);
234                     }else if ( val == "" ){
235                         option.list.hide();
236                     }
237                 }, 300);
238             }
239             // 不需要检索列表
240         } else{
241             val = $(this).val();
242             if (val != "") {
243                 if (keyCode == keys.enter) {
244                     // option.createChosen($(this), val);
245                     return false;
246                 }
247             }
248         };
249     });
250 
251     $text.off("keydown").on("keydown", function(event){
252         var keyCode = event.keyCode;
253         if (keyCode == keys.back) {
254             if( $.trim( $(this).val() ) === ""){
255                 if($list.is(":visible")){ $list.hide(); }
256             }
257         }
258     });
259 
260     option.autoBtn.on("click", function(){
261         $list.hide();
262         if ( typeof( option.action ) == "function" ) {
263             option.action(option, $.trim( $text.val() ) );
264         }
265     });
266 }
