 40                     '<div class="popup-btn-wrap Js-popup-btn-wrap">'+

 42                         '<a class="popup-btn Js-popup-cancel" href="javascript:;;"><span class="popup-btn-text">取消</span></a>'+
 43                     '</div>'+
 44                 '</div>'),
 45         template : "<div></div>",
 46   [可选]addClass : "popupClass",
 47   [可选]isLayer : true,
 48   [可选]isCenter : true,
 49   [可选]isDrag : true,
 50   [可选]isOnly : false,
 51   [可选]autoShow : true,
 52   [可选]append : {
 53             isAppend : false,
 54             dom : $(".dom")
 55         },
 56   [可选]hasBtn : true,
 57   [可选]hasCancel : true,

 60   [可选]cancel : null || function(opt){}
 61     }

 66         type: "confirm",
 67   [可选]param : {},
 68   [可选]title: "提示",

 70   [可选]cancelText : "确定按钮",
 71         message : "测试弹出框",
 72   [可选]done : function(opt){},
 73   [可选]cancel : function(opt){}

 77     {
 78         type: "result",
 79   [可选]param : {},
 80   [可选]title: "提示",

 89 $.cgPopup = function(options){
 90     var $layer, $popup, $close, $title, $con, $error, $btnWrap, $done, $cancel,
 91         option = $.extend(/** @lends jQuery.cgPopup.prototype*/{
 92             /**
 93              * 弹出框的种类
 94              * @type String
 95              * @default "popup"
 96              */
 97             type : "popup",
 98             /**
 99              * 附加参数
100              * @type Object
101              * @default {}
102              */
103             param : {},
104             /**
105              * 弹出框的标题
106              * @type String
107              * @default ""
108              */
109             title: "",
110             /**
111              * 确定按钮的文字
112              * @type String
113              * @default "确定"
114              */
115             doneText: "确定",
116             /**
117              * 取消按钮的文字
118              * @type String
119              * @default "取消"
120              */
121             cancelText: "取消",
122             /**
123              * 弹出框html的jQuery对象
124              * @type jQuery Dom
125              * @default null
126              */
127             popupTemp : null,
128             /**
129              * 内容区的html代码片段
130              * @type html String
131              * @default ""
132              */
133             template : "",
134             /**
135              * 弹出框为展示文本类型时显示的消息提示文本
136              * @type String
137              * @default ""
138              */
139             message : "",
140             /**
141              * 如果type = result 自动消失时间
142              * @type Number
143              * @default null
144              */
145             time : null,
146             /**
147              * 附加弹出框样式
148              * @type css class
149              * @default ""
150              */
151             addClass : "",
152             /**
153              * 是否需要遮罩层
154              * @type Boolean
155              * @default true
156              */
157             isLayer : true,
158             /**
159              * 是否居中
160              * @type Boolean
161              * @default true
162              */
163             isCenter : true,
164             /**
165              * 是否唯一
166              * @type Boolean
167              * @default true
168              */
169             isDrag : true,
170             /**
171              * 是否唯一
172              * @type Boolean
173              * @default false
174              */
175             isOnly : false,
176             /**
177              * 是否在初始化时自动显示
178              * @type Boolean
179              * @default true
180              */
181             autoShow : true,
182             /**
183              * 以添加的方式初始化弹出框 isAppend: 是否以添加的方式初始化弹出框; dom: 包裹对象
184              * @type Object
185              * @default isAppend : false, dom : $(".dom")
186              */
187             append : {
188                 isAppend : false,
189                 dom : $(".dom")
190             },
191             /**
192              * 是否需要按钮
193              * @type Boolean
194              * @default true
195              */
196             hasBtn : true,
197             /**
198              * 是否需要取消按钮
199              * @type Boolean
200              * @default true
201              */
202             hasCancel : true,
203             /**
204              * 内容区的附加函数
205              * @type Function
206              * @param {Object} opt 外部调用对象
207              * @default null
208              */
209             content :null,
210             /**
211              * 确定按钮附加函数
212              * @type Function
213              * @param {Object} opt 外部调用对象
214              * @default null
215              */
216             done :null,
217             /**
218              * 取消按钮附加函数
219              * @type Function
220              * @param {Object} opt 外部调用对象
221              * @default null
222              */
223             cancel :null
224         }, options),
225         targetContent = option.content,
226         targetDone = option.done,
227         targetCancel = option.cancel,
228         boolContent = typeof option.content == "function" ? true : false,
229         boolDone = typeof option.done == "function" ? true : false,
230         boolCancel = typeof option.cancel == "function" ? true : false;
231 
232     $popup = option.popupTemp || $('<div class="module-popup fn-clear Js-popup-wrap">'+
233         '<a href="javascript:void(0)" class="popup-close Js-popup-close"></a> '+
234         '<h6 class="fn-clear popup-tit Js-popup-title">' + option.title + '</h6>'+
235         '<div class="popup-con Js-popup-con"></div>'+
236         '<div class="popup-btn-wrap Js-popup-btn-wrap">'+
237             '<a class="popup-btn done Js-popup-done" href="javascript:;;"><span class="popup-btn-text">' + option.doneText + '</span></a>'+
238             '<a class="popup-btn cancel Js-popup-cancel" href="javascript:;;"><span class="popup-btn-text">' + option.cancelText + '</span></a>'+
239         '</div>'+
240     '</div>');
241 
242     $error = $('<p class="confirm-msg"></p>');
243 
244     $title = $popup.find(".Js-popup-title");
245     $close = $popup.find(".Js-popup-close");
246     $con = $popup.find(".Js-popup-con");
247     $btnWrap = $popup.find(".Js-popup-btn-wrap");
248     $done = $btnWrap.find(".Js-popup-done");
249     $cancel = $btnWrap.find(".Js-popup-cancel");
250 
251     $.extend(option, {
252         oPopup : $popup,
253         oCon : $con,
254         oBtnWrap : $btnWrap,
255         oBtnDone : $done,
256         oBtnCancel : $cancel,
257         oBtnClose : $close,
258         oLayer : $layer,
259         /**
260          * @name jQuery.cgPopup#bindDrag
261          * @desc  绑定拖拽事件
262          * @event
263          */
264         bindDrag : function(){
265             var _drag = {},
266                 width = $popup.width(),
267                 height = $popup.height();
268             // bind drag
269             $title.css("cursor", "move");
270             $popup.on("mousedown", ".Js-popup-title", function(event) {
271                 event.preventDefault();
272                 var _position = $popup.position();
273                 _drag.posLeft = _position.left;
274                 _drag.posTop = _position.top - $(window).scrollTop();
275                 _drag.dl = event.pageX;
276                 _drag.dt = event.pageY;
277                 $(document).on("mousemove", function(e){
278                     e.preventDefault();
279                     _drag.ml = e.pageX;
280                     _drag.mt = e.pageY;
281                     _drag.ol = _drag.posLeft + _drag.ml - _drag.dl;
282                     _drag.ot = _drag.posTop + _drag.mt - _drag.dt;
283                     $popup.css({
284                         left : _drag.ol + "px",
285                         top : _drag.ot + "px"
286                     });
287                 });
288                 $(document).one("mouseup", function(){
289                     $(document).off("mousemove mouseup");
290                 });
291             });
292         },
293         /**
294          * @name jQuery.cgPopup#positionCenter
295          * @desc  使弹出框居中
296          * @event
297          */
298         positionCenter : function(){
299             var width = $popup.width(),
300                 height = $popup.height();
301             $popup.css({
302                 "position" : "fixed",
303                 "top" : "40%",
304                 "left" : "50%",
305                 "margin" : "-" + height / 2 + "px 0 0 -" + width / 2 + "px" 
306             });
307         },
308         /**
309          * @name jQuery.cgPopup#show
310          * @desc  显示弹出框
311          * @event
312          * @param {Function} callback 回调函数
313          */
314         show : function(callback){
315             option.disableBtn(true);
316             option.positionCenter();
317             if ( option.isLayer ) {
318                 $layer.show();
319             }
320             $popup.show();
321             if (Modernizr.csstransitions) {
322                 $popup.addClass("fadeInDown").delay(800).show(0, function(){$(this).removeClass("fadeInDown");});
323             }
324             if ( typeof callback == "function" ) {
325                 callback(option);
326             }
327         },
328         /**
329          * @name jQuery.cgPopup#hide
330          * @desc  隐藏弹出框
331          * @event
332          * @param {Function} callback 回调函数
333          */
334         hide : function(callback){
335             if ( option.isLayer ) {
336                 $layer.hide();
337             }
338             if (!Modernizr.csstransitions) {
339                 $popup.hide();
340             }else{
341                 $popup.addClass("fadeOutUp").delay(800).hide(0, function(){$(this).removeClass("fadeOutUp");});
342             }
343             if ( typeof callback == "function" ) {
344                 callback(option);
345             }
346         },
347         /**
348          * @name jQuery.cgPopup#close
349          * @desc  关闭弹出框
350          * @event
351          * @param {Function} callback 回调函数
352          */
353         close : function(callback){
354             if ( option.isLayer ) {
355                 $layer.remove();
356             }
357             if (!Modernizr.csstransitions) {
358                 $popup.remove();
359             }else{
360                 $popup.addClass("fadeOutUp").delay(700).hide(0, function(){
361                     $popup.remove();
362                 });
363             }
364             if ( typeof callback == "function" ) {
365                 callback(option);
366             }
367         },
368         /**
369          * @name jQuery.cgPopup#showTip
370          * @desc  显示错误信息
371          * @event
372          * @param {String} str 错误信息的字符串
373          */
374         showTip : function(str){
375             $popup.addClass("rock").delay(500).show("0", function(){
376                 $(this).removeClass("rock");
377                 option.disableBtn(true);
378             });
379             if ( !$con.find($error).length ) {
380                 $con.append($error);
381             }
382             $error.show().html(str);
383         },
384         /**
385          * @name jQuery.cgPopup#removeTip
386          * @desc  隐藏错误信息
387          * @event
388          */
389         removeTip : function(){
390             $error.hide();
391         },
392         /**
393          * @name jQuery.cgPopup#reset
394          * @desc  重置内容区内容
395          * @event
396          * @param {html String} newHtml 重置内容区的html，但是事件不会重置
397          */
398         reset : function(newHtml){
399             if ( !!newHtml ) {
400                 $con.html(newHtml);
401             } else{
402                 $con.html(option.template);
403             }
404         },
405         /**
406          * @name jQuery.cgPopup#disableBtn
407          * @desc  改变确定按钮状态
408          * @event
409          * @param {Boolean} isReset 为true时确定按钮可用，反之不可用
410          */
411         disableBtn : function(isReset){
412             var self = this;
413             if ( isReset ) {
414                 $done.removeClass("popup-btn-disabled");
415             } else{
416                 $done.addClass("popup-btn-disabled");
417             }
418         }
419     });
420 
421     // 派生多种类型
422     switch(option.type){
423         case "popup":
424         break;
425         case "confirm":
426             if ( !option.message ) {
427                 return false;
428             }
429             option.title = option.title || "提示";
430             option.template = '<p class="confirm-msg">' + option.message + '</p>';
431             if ( boolDone ) {
432                 option.done = function(){
433                     targetDone(option);
434                     option.close();
435                 }
436             }
437         break;
438         case "result":
439             if ( !option.message ) {
440                 return false;
441             }
442             option.title = option.title || "提示";
443             if ( typeof option.message == "string" ) {
444                 option.template = '<p class="confirm-msg">' + option.message + '</p>';
445             }
446             if ( option.message.constructor == Array && option.message.length ) {
447                 var _str = "";
448                 for( var i = 0; i < option.message.length; i++ ){
449                     _str += '<p class="confirm-error-msg">' + option.message[i] + '</p>';
450                 }
451                 option.template = _str;
452             }
453             option.hasCancel = false;
454             if ( !!option.time ) {
455                 if ( boolContent ) {
456                     targetContent(option);
457                 }
458                 option.content = function(){
459                     setTimeout(function(){
460                         option.close();
461                     }, option.time);
462                 }
463             }
464         break;
465         default:
466         break;
467     }
468 
469     if ( !option.title ) {
470         $title.hide();
471     }
472 
473     if ( option.isLayer ) {
474         $layer = $("<div class='module-popup-layer'></div>");
475         $("body").append($layer);
476     }
477 
478     if ( option.isOnly ) {
479         $(".module-popup-layer").remove();
480         $(".Js-popup-wrap").remove();
481     }
482 
483     if ( option.append.isAppend ) {
484         option.append.dom.append( $popup );
485     } else{
486         $("body").append( $popup );
487     }
488 
489     // add new class
490     if ( option.addClass ) {
491         $popup.addClass(option.addClass);
492     }
493 
494     // append content template
495     $con.append(option.template);
496 
497     // bind popup init function
498     if ( typeof(option.content) == "function" ) {
499         option.content( option );
500     }
501     // bind btn close method
502     $popup.on("click", ".Js-popup-close", function() {
503         if ( option.hasCancel && boolCancel ) {
504             option.cancel( option );
505         } else{
506             if ( !option.hasCancel && boolDone ) {
507                 option.done( option );
508             }
509             option.close();
510         }
511     // bind btn done method
512     }).on("click", ".Js-popup-done", function() {
513         if ( option.hasBtn && !$(this).hasClass("popup-btn-disabled") && boolDone ) {
514             option.disableBtn();
515             option.done( option );
516         }
517         if ( !boolDone ) {
518             option.close();
519         }
520         return false;
521     // bind btn cancel method
522     }).on("click", ".Js-popup-cancel", function() {
523         if ( boolCancel ) {
524             option.cancel( option );
525         }else{
526             option.close();
527         }
528     });
529 
530     // bind drag and drop
531     if ( option.isDrag ) {
532         option.bindDrag();
533     }
534 
535     // change button
536     if ( option.hasBtn ) {
537         if ( !option.hasCancel ) {
538             $cancel.remove();
539         }
540     }else{
541         $btnWrap.remove();
542     }
543 
544     // bind wrap position
545     if ( option.isCenter && !option.append.isAppend ) {
546         $(window).resize(function(){
547             option.positionCenter();
548         });
549     }
550 
551     // change auto show
552     if ( option.autoShow ) {
553         option.show();
554     }
555 
556     return option;
557 };

560 }(jQuery));
