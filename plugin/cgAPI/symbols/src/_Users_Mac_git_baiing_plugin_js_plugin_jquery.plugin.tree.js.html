1 (function($){
  2 
  3 /**
  4  * @author 徐晨 
  5  * @name jQuery.cgTree
  6  * @class 自动完成插件 <a href="../demo/tree.html" target="_blank">demo</a>
  7  * @constructor
  8  * @extends jQuery
  9  * @extends jQuery.getFromTemplate
 10  * @since version 1.0 
 11  * @param {Object} options 参数对象数据
 12  * @param {jQuery Object} options.wrap 包裹的jquery dom
 13  * @param {json} options.treeData 树形数据 {"id":"id","pId":"pId|null","name":"描述型知识"(,"prefix":"c")}
 14  * @param {css class} options.addClass 点击节点时需要添加的样式
 15  * @param {Boolean} options.isPreFix 是否存在特殊识别符
 16  * @param {jQuery Object} options.template 节点模板对象（按照jQuery.getFromTemplate方法）
 17  * @param {Function} options.nodeFun 点击节点时的附加事件
 18  * @param {Function} options.addOtherFun 添加非末级节点时的附加事件
 19  * @param {Function} options.addLastFun 添加末级节点时的附加事件
 20  * @param {Function} options.callback 模块初始化时的附加事件
 21  * @example
 22  * $.cgTree({
 23         wrap : $(".Js-has-prefix"),
 24         treeData : [{"id":"c102","pId":"c97","prefix":"c","name":"描述型知识","type":"catalog"}],
 25         template : $("#Js-tree-item-template"),
 26         isPreFix : true,
 27         callback : function (opt) {
 28         }
 29     });
 30  * @example
 31  *  <ul class="module-tree-list Js-has-prefix"></ul>
 32     <script id="Js-tree-item-template" type="application/template">
 33         <li class="Js-tree-item li-item fn-clear #{status}" style="padding-left:#{padding}px; border-bottom:1px solid #EEE;" data-id="#{newID}" data-lv="#{lv}" data-prefix="#{prefix}">
 34             <span class="tree-icon" style="left:#{padding}px">
 35                 <img src="images/lib/i.gif" class="icon-lv Js-tree-icon" />
 36             </span>
 37             <a href="#{url}" class="name Js-tree-link">#{name}</a>
 38         </li>
 39     </script>
 40  */
 41 $.cgTree = function( options ) {
 42     var option = $.extend({
 43         wrap : $(".tree-list"),
 44         treeData : [],
 45         addClass : "",
 46         isPreFix : true,
 47         template : null,
 48         nodeFun : null,
 49         addOtherFun : null,
 50         addLastFun : null,
 51         callback : null
 52     }, options);
 53     var _arrTree = { root : [] },
 54         _thisTreeData = [],
 55         _strTemp = $('<div><li style="margin-left:#{padding}px" id="color_#{id}" class="Js-tree-item #{status}" data-id="#{id}" data-lv="#{lv}" data-pid="#{pId}">'+
 56                         '<a href="javascript:void(0)"  class="tree-link Js-tree-link" data-id="#{id}">#{name}</a>'+
 57                     '</li></div>'),
 58         _temp = option.template ? option.template : _strTemp,
 59         _createTree = function(item){
 60             var $item;
 61             if ( _arrTree[item.id] ) {
 62                 item.hasChild = true;
 63                 item.status = "on"; 
 64                 $item = option.addItem(item);
 65                 if ( typeof(option.addOtherFun) == "function" ) {
 66                     option.addOtherFun($item, item);
 67                 }
 68                 $.each(_arrTree[item.id], function(i, t){
 69                     t.lv = Math.floor(item.lv) + 1;
 70                     _createTree(t);
 71                 });
 72             }else{
 73                 item.hasChild = false;
 74                 item.status = "last";
 75                 $item = option.addItem(item);
 76                 if ( typeof(option.addLastFun) == "function" ) {
 77                     option.addLastFun($item, item);
 78                 }
 79             }
 80         };
 81     $.extend(option, {
 82         /**
 83          * @name jQuery.cgTree#addItem
 84          * @desc  添加一个节点
 85          * @event
 86          * @param {object} item 节点的json数据
 87          * @return {jQuery Object} [ 被添加节点的jQuery对象]
 88          */
 89         addItem : function(item){
 90 
 91             item.padding = item.lv * 20;
 92             item.newID = item.id;
 93             if ( option.isPreFix ) {
 94                 var _r = new RegExp( "^" + item.prefix , "gi" );
 95                 item.newID = item.newID.replace(_r, "");
 96             }else{
 97                 item.prefix = undefined;
 98             }
 99             var $li = $.getFromTemplate( _temp, item );
100             option.wrap.append( $li );
101             return $li;
102         },
103         /**
104          * @name jQuery.cgTree#openRoot
105          * @desc  打开所有顶级节点，并关闭其他节点
106          * @event
107          */
108         openRoot : function(){
109             option.wrap.find(".Js-tree-item").hide().filter(".on").removeClass("on").addClass("off");
110             $.each(_arrTree.root, function(index, item){
111                 option.wrap.find(".Js-tree-item[data-id='" + item.newID + "'][data-prefix='" + item.prefix + "']").show().removeClass("off").addClass("on");
112                 if ( !!_arrTree[item.id] ) {
113                     $.each(_arrTree[item.id], function(i, t){
114                         option.wrap.find(".Js-tree-item[data-id='" + t.newID + "'][data-prefix='" + t.prefix + "']").show().filter(".on").removeClass("on").addClass("off");
115                     });
116                 }
117             });
118         },
119         /**
120          * @name jQuery.cgTree#openSelected
121          * @desc  打开当前节点
122          * @event
123          * @param {String} id 要打开的节点id
124          * @param {String} prefix 要打开的节点id的特殊识别符
125          */
126         openSelected : function(id, prefix){
127             // console.log(id, prefix);
128             var _id = id;
129             if ( !!prefix ) {
130                 id = prefix + id;
131             }
132             if ( !!_arrTree[id] ){
133                 if ( !!prefix ) {
134                     option.wrap.find(".Js-tree-item[data-id='" + _id + "'][data-prefix='" + prefix + "']").removeClass("off").addClass("on");
135                     $.each(_arrTree[id], function(index, item){
136                         option.wrap.find(".Js-tree-item[data-id='" + item.newID + "'][data-prefix='" + item.prefix + "']").show();
137                     });
138                 }else{
139                     option.wrap.find(".Js-tree-item[data-id='" + _id + "']").removeClass("off").addClass("on");
140                     $.each(_arrTree[id], function(index, item){
141                         option.wrap.find(".Js-tree-item[data-id='" + item.newID + "']").show();
142                     });
143                 }
144             }
145         },
146         /**
147          * @name jQuery.cgTree#closeSelected
148          * @desc  关闭当前节点
149          * @event
150          * @param {String} id 要关闭的节点id
151          * @param {String} prefix 要关闭的节点id的特殊识别符
152          */
153         closeSelected : function(id, prefix){
154             // console.log(id, prefix);
155             var _id = id;
156             if ( !!prefix ) {
157                 id = prefix + id;
158             }
159             // console.log(id);
160             if ( _arrTree[id] ){
161                 if ( !!prefix ) {
162                     option.wrap.find(".Js-tree-item[data-id='" + _id + "'][data-prefix='" + prefix + "']").removeClass("on").addClass("off");
163                     $.each(_arrTree[id], function(index, item){
164                         option.wrap.find(".Js-tree-item[data-id='" + item.newID + "'][data-prefix='" + item.prefix + "']").hide();
165                         var _closeID = item.id;
166                         if ( option.isPreFix ) {
167                             var _r = new RegExp( "^" + item.prefix , "gi" );
168                             _closeID = _closeID.replace(_r, "");
169                         }
170                         option.closeSelected(_closeID, item.prefix);
171                     });
172                 }else{
173                     option.wrap.find(".Js-tree-item[data-id='" + _id + "']").removeClass("on").addClass("off");
174                     $.each(_arrTree[id], function(index, item){
175                         option.wrap.find(".Js-tree-item[data-id='" + item.newID + "']").hide();
176                         var _closeID = item.id;
177                         if ( option.isPreFix ) {
178                             var _r = new RegExp( "^" + item.prefix , "gi" );
179                             _closeID = _closeID.replace(_r, "");
180                         }
181                         option.closeSelected(_closeID, item.prefix);
182                     });
183                 }
184             }
185         }
186     });
187     // parse data
188     _thisTreeData = [].concat(option.treeData);
189     $.each(_thisTreeData, function(index, item){
190         // console.log(item);
191         if( (item.open != undefined && item.open) || item.pId == "null" || item.pId == null ){
192             item.lv = 0;
193 
194             _arrTree.root.push(item);
195         }else{
196             if (!_arrTree[item.pId]) {
197                 _arrTree[item.pId] = [];
198             }
199             _arrTree[item.pId].push(item);
200         }
201     });
202     // console.log(_arrTree.root);
203     // init create tree
204     $.each(_arrTree.root, function(index, item){
205         _createTree(item);
206     });
207     // bind click fun
208     option.wrap.on("click", ".Js-tree-item", function(event){
209         var $item = $(this),
210             _id = $item.data("id"),
211             _preFix = $item.data("prefix"),
212             $element = $(event.target);
213         if ( $element.hasClass("Js-tree-link") || $element.hasClass("Js-tree-icon") ) {
214             if ( option.addClass != "" ) {
215                 option.wrap.children(".cur").removeClass( option.addClass );
216                 $item.addClass( option.addClass );
217             }
218             if ( $element.hasClass("Js-tree-link") && typeof(option.nodeFun) == "function" )  {
219                 option.nodeFun($item, option, _id);
220                 // return false;
221             }
222             if ( $item.hasClass("off") ) {
223                 if ( option.isPreFix ) {
224                     option.openSelected(_id, _preFix);
225                 } else{
226                     option.openSelected(_id);
227                 }
228             } else if ( $item.hasClass("on") ) {
229                 if ( option.isPreFix ) {
230                     option.closeSelected(_id, _preFix);
231                 } else{
232                     option.closeSelected(_id);
233                 }
234             }
235         }
236     });
237     if ( typeof(option.callback) == "function" ) {
238         option.callback(option);
239     }
240 };
241 }(jQuery));
